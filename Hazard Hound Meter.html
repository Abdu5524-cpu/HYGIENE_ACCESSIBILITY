// src/App.jsx

import { useState, useEffect } from 'react';

import './App.css';


function App() {

  const [hazardData, setHazardData] = useState({

    fire: 0,

    gas: 1,

    structural: 3,

    air: 10,

    temp: 5,

    emf: 0,

    touch: 0

  });


  const [timestamp, setTimestamp] = useState(new Date());


  useEffect(() => {

    const timer = setInterval(() => {

      setTimestamp(new Date());

    }, 1000);

    return () => clearInterval(timer);

  }, []);


  const calculateTotal = () => {

    return hazardData.fire + hazardData.gas + hazardData.structural + 

           hazardData.air + hazardData.temp + hazardData.emf + hazardData.touch;

  };


  const total = calculateTotal();

  const statusText = total < 10 ? "✓ SAFE" : total < 30 ? "⚠️ CAUTION" : "🔴 DANGER";

  const statusClass = total < 10 ? "safe" : total < 30 ? "warning" : "danger";


  const hazards = [

    { name: "FIRE", key: "fire", icon: "🔥", color: "fire" },

    { name: "GAS", key: "gas", icon: "💨", color: "gas" },

    { name: "STRUCTURAL", key: "structural", icon: "🏗️", color: "structural" },

    { name: "AIR QUALITY", key: "air", icon: "🌫️", color: "air" },

    { name: "INTERNAL TEMP", key: "temp", icon: "🌡️", color: "temp" },

    { name: "EMF/EMI", key: "emf", icon: "⚡", color: "emf" },

    { name: "TOUCH", key: "touch", icon: "🖐️", color: "touch" }

  ];


  return (

    <div className="dashboard">

      <div className="header">

        <h1>⚠️ HAZARD HOUND</h1>

        <div className="subtitle">7 Hazard Detection | Fire • Gas • Structural • Air • Temp • EMF • Touch</div>

      </div>


      <div className="main-status">

        <div className="percentage">

          <span>{total}</span><span className="small">%</span>

        </div>

        <div className={`status-label ${statusClass}`}>{statusText}</div>

        <div className="timestamp">

          Updated: {timestamp.toLocaleTimeString('en-US', { hour12: true })}

        </div>

      </div>


      <div className="hazard-grid">

        {hazards.map((hazard) => (

          <div className="hazard-item" key={hazard.key}>

            <div className="hazard-icon">{hazard.icon}</div>

            <div className="hazard-info">

              <div className="hazard-name">{hazard.name}</div>

              <div className="hazard-bar-container">

                <div 

                  className={`hazard-bar bar-${hazard.color}`} 

                  style={{ width: `${hazardData[hazard.key]}%` }}

                ></div>

              </div>

            </div>

            <div className="hazard-percent">{hazardData[hazard.key]}%</div>

          </div>

        ))}

      </div>


      <div className="footer">Hazard Hound v1.0 • Real-time monitoring</div>


      <style jsx>{`

        * { margin: 0; padding: 0; box-sizing: border-box; }

        body { font-family: -apple-system, BlinkMacSystemFont, 'Segoe UI', Roboto, sans-serif; background: linear-gradient(135deg, #0a0f1e 0%, #0c1222 100%); min-height: 100vh; padding: 2rem; color: #fff; }

        .dashboard { max-width: 500px; margin: 0 auto; background: rgba(18, 25, 45, 0.8); backdrop-filter: blur(10px); border-radius: 2rem; padding: 1.5rem; box-shadow: 0 25px 45px rgba(0,0,0,0.3); }

        .header { text-align: center; margin-bottom: 2rem; }

        .header h1 { font-size: 1.8rem; font-weight: 700; background: linear-gradient(135deg, #fff, #a0a8c0); -webkit-background-clip: text; background-clip: text; color: transparent; }

        .subtitle { font-size: 0.75rem; color: #6c7a9e; }

        .main-status { background: linear-gradient(135deg, #1a2542, #0f172a); border-radius: 1.5rem; padding: 1.5rem; text-align: center; margin-bottom: 1.5rem; border: 1px solid rgba(72, 187, 120, 0.3); }

        .percentage { font-size: 4rem; font-weight: 800; }

        .percentage .small { font-size: 1.5rem; }

        .status-label { font-size: 1.2rem; font-weight: 600; margin-top: 0.5rem; }

        .safe { color: #48bb78; }

        .warning { color: #f6ad55; }

        .danger { color: #fc8181; }

        .timestamp { font-size: 0.7rem; color: #6c7a9e; margin-top: 0.75rem; }

        .hazard-grid { display: flex; flex-direction: column; gap: 0.75rem; }

        .hazard-item { background: rgba(15, 23, 42, 0.6); border-radius: 1rem; padding: 0.75rem 1rem; display: flex; align-items: center; gap: 1rem; border: 1px solid rgba(255,255,255,0.05); }

        .hazard-icon { width: 3rem; font-size: 1.5rem; text-align: center; }

        .hazard-info { flex: 1; }

        .hazard-name { font-weight: 600; font-size: 0.85rem; color: #a0aec0; }

        .hazard-bar-container { margin-top: 0.25rem; height: 0.5rem; background: rgba(255,255,255,0.1); border-radius: 1rem; overflow: hidden; }

        .hazard-bar { height: 100%; border-radius: 1rem; transition: width 0.3s ease; }

        .bar-fire { background: linear-gradient(90deg, #f56565, #fc8181); }

        .bar-gas { background: linear-gradient(90deg, #ed8936, #f6ad55); }

        .bar-structural { background: linear-gradient(90deg, #ecc94b, #f6e05e); }

        .bar-air { background: linear-gradient(90deg, #4299e1, #63b3ed); }

        .bar-temp { background: linear-gradient(90deg, #9f7aea, #b794f4); }

        .bar-emf { background: linear-gradient(90deg, #38b2ac, #4fd1c5); }

        .bar-touch { background: linear-gradient(90deg, #ed64a6, #f687b3); }

        .hazard-percent { font-size: 0.9rem; font-weight: 600; min-width: 2.5rem; text-align: right; color: #cbd5e0; }

        .footer { margin-top: 1.5rem; text-align: center; font-size: 0.7rem; color: #4a5568; }

        @media (max-width: 550px) { body { padding: 1rem; } .percentage { font-size: 3rem; } }

      `}</style>

    </div>

  );

}


export default App;
